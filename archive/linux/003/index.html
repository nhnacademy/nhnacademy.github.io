#!/usr/bin/env bash

defined() {
    [ $# -lt 1 ] && exit 1 
    for var in "$@"; do
        if [ -z "${!var:-}" ]; then
            error "$var is undefined"
            exit 1
        fi
    done
    exit 0
}
